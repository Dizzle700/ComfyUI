# Скопируйте и вставьте эту команду в поле "Docker Command" или "Startup Command" в настройках пода на RunPod:

bash -c "cd /workspace && ( [ -f runpod_startup.sh ] || git clone https://github.com/Dizzle700/ComfyUI.git . ) && bash runpod_startup.sh"

# Инструкция:
# 1. Перед запуском убедитесь, что последние изменения запушены в репозиторий.
# 2. Не забудьте открыть HTTP Ports в настройках RunPod:
#    - Port 7860 (для Панели Управления)
